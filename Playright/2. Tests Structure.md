### Locator Syntax Rules

```js
test('Locator syntax rules', async ({page}) => {
    //by Tag name
    await page.locator('input').first().click();

    //by ID
    page.locator("#inputEmail");

    //by class value
    page.locator(".shape-rectangle");

    //by attribute
    page.locator('[placeholder="Email"]');

    //by full class
    page.locator('[class="input-full-width size-medium status-basic shape-rectangle nb-transition"]');

    //by combine different selectors
    page.locator('input[placeholder="Email"][nbinput]')

    //by xPath(not recommended)
    page.locator('//*[@id="inputEmail1"]');

})
```

### User Facing locators

```js

test('User facing locators', async({page})=> {
    await page.getByRole("textbox", {name: "Email"}).first().click();

    await page.getByRole("button", {name: "Sign in"}).first().click();

    await page.getByLabel('Email').first().click();

    await page.getByPlaceholder('Jogn Doe').click();

    await page.getByText('Using the Grid').click();

    await page.getByTestId("");
```

### Locating child elements

```js
test('locating child elements', async ({page}) => {
    await page.locator('ng-card nb-radio :text-is("Option 1")').click();

    await page.locator('ng-card').locator('ng-radio').locator('text-is("Option 1")').first().click();

    await page.locator('nb-card').getByRole("button", {name: "Sign in"}).first().click();

    await page.locator('ng-card').nth(3).getByRole('button').click(); 
    // not recommended to pick by index
})
```


### Reusing Locators

```js

test('locating the parents', async ({page}) => {
    await page.locator("nb-card", {hasText: "Using the Grid"}).getByRole('textbox', {name: "Email"}).click();

    await page.locator("nb-card", {has: page.locator("#inputEmail")}).getByRole("textbox", {name: "Email"}).click();

    await page.locator('nb-card').filter({hasText: "Basic form"}).getByRole('textbox', {name: "Email"}).click();

    await page.locator('nb-card').filter({has: page.locator('.status-danger')}).getByRole('textbox', {name: "Password"}).click();

    await page.locator('ng-card').filter({has: page.locator('ng-checkbox')}).filter({hasText: "Sign in"})
    .getByRole('textbox', {name: "Email"}).click();
```
