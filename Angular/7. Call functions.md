### How to value from input and display on screen (Binding)



app.compnent.html
```js
<input type="text" placeholder="enter name" #fname (keyup)="handleChange(fname.value)"/>

<p>{{displayValue}}</p>
```

```js
export class AppComponent {
  title = 'thirdproject';

  displayValue = "";

  handleChange(val: any){
    console.log(val);
    this.displayValue = val;
  }
}
```
