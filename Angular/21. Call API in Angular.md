### Post API
in userData.service.ts


```js
export class UserdataService {
  constructor(private http: HttpClient) { }

  url: any = "https://jsonplaceholder.typicode.com/todos"
  members() {
    return this.http.get(this.url);
  }
  saveMember(data:any) {
    return this.http.post(this.url,data)
}
```
